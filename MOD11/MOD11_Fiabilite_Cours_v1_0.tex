\section{Vocabulaire de la fiabilité \hfill {\normalsize\color{uimmred}[C01]}}

\subsection{Définition et grandeurs}

\begin{definition}
La \textbf{fiabilité} est l'aptitude d'un dispositif à accomplir une fonction requise, dans des conditions données, pendant une durée donnée (norme NF X06-501). Mathématiquement, c'est la \textbf{probabilité de non-défaillance} sur une durée $t$.
\end{definition}

\begin{definition}
On note $T$ la variable aléatoire ``temps de bon fonctionnement'' (TBF). La \textbf{fonction de fiabilité} est :
\[
\boxed{R(t) = P(T > t)}
\]
et la \textbf{fonction de défaillance} (ou de répartition) :
\[
F(t) = P(T \leq t) = 1 - R(t).
\]
\end{definition}

\begin{definition}
La \textbf{densité de défaillance} est $f(t) = F'(t) = -R'(t)$. Le \textbf{taux de défaillance instantané} est :
\[
\boxed{\lambda(t) = \frac{f(t)}{R(t)} = \frac{-R'(t)}{R(t)}}
\]
Il représente la proportion de dispositifs survivants qui défaillent par unité de temps. Son unité est l'inverse d'un temps (par exemple panne/heure).
\end{definition}

\begin{remarque}
$\lambda(t)$ s'estime, sur un intervalle $[t\,;\,t+\Delta t]$, par :
\[
\lambda(t) \approx \frac{\text{nombre de défaillants pendant }\Delta t}{(\text{nombre de survivants au début})\times \Delta t}.
\]
\end{remarque}

\subsection{Temps caractéristiques : MTBF, MTTF, MTTR}

\begin{definition}
\begin{itemize}
  \item \textbf{MTBF} (\textit{Mean Time Between Failures}) : temps moyen \textbf{entre} deux défaillances d'un système \textbf{réparable}.
  \[
  \text{MTBF} = \frac{\text{durée totale de bon fonctionnement}}{\text{nombre de défaillances}} = \int_0^{+\infty} R(t)\,dt.
  \]
  \item \textbf{MTTF} (\textit{Mean Time To Failure}) : temps moyen \textbf{avant} la première défaillance d'un dispositif \textbf{non réparable}.
  \item \textbf{MTTR} (\textit{Mean Time To Repair}) : temps moyen de réparation.
\end{itemize}
\end{definition}

\begin{exemple}[-- Calcul de MTBF]
Une ligne fonctionne $130$\,h par semaine avec $4$ pannes immobilisant respectivement $2, 2, 3, 3$ heures.
\[
\text{MTBF} = \frac{130 - (2+2+3+3)}{4} = \frac{120}{4} = 30\ \text{heures.}
\]
Le taux de défaillance moyen vaut $\lambda = \dfrac{1}{\text{MTBF}} = \dfrac{1}{30} \approx 0{,}033$ panne/h.
\end{exemple}

\subsection{Maintenabilité et disponibilité}

\begin{definition}
La \textbf{maintenabilité} $M(t)$ est la probabilité que la réparation soit achevée avant la durée $t$. La \textbf{disponibilité} $D$ (en régime permanent) est la proportion de temps pendant laquelle le système est opérationnel :
\[
\boxed{D = \frac{\text{MTBF}}{\text{MTBF} + \text{MTTR}}}
\]
\end{definition}

\begin{exemple}
Un équipement a une MTBF de $30$\,h et une MTTR de $2$\,h. Sa disponibilité est :
\[
D = \frac{30}{30 + 2} = \frac{30}{32} \approx 0{,}94 \;\; (94\,\%).
\]
\end{exemple}

\begin{remarque}
On parle de démarche \textbf{F.M.D.} : \textbf{F}iabilité (ne pas tomber en panne), \textbf{M}aintenabilité (être réparé vite), \textbf{D}isponibilité (être utilisable). La disponibilité résulte des deux premières.
\end{remarque}

\subsection{La courbe en baignoire}

Le taux de défaillance d'un matériel évolue typiquement en trois phases.

\section{Estimation à partir de données \hfill {\normalsize\color{uimmred}[C02]\,[C04]}}

\subsection{Méthode des classes (grand échantillon)}

\begin{methode}[-- Taux de défaillance empirique]
Pour chaque intervalle $\Delta t$ :
\[
\lambda(t) = \frac{n(\Delta t)}{N_{\text{début}} \times \Delta t}
\]
où $n(\Delta t)$ est le nombre de défaillants et $N_{\text{début}}$ le nombre de survivants au début de l'intervalle. La fiabilité $R$ s'obtient par le rapport des survivants à l'effectif initial.
\end{methode}

\begin{exemple}[-- Doigts de préhension]
Sur $14$ machines, on relève sur l'intervalle $0$--$150$\,h une défaillance :
\[
\lambda = \frac{1}{14 \times 150} \approx 4{,}8 \times 10^{-4}\ \text{panne/h.}
\]
On répète le calcul sur chaque classe pour tracer $\lambda(t)$ et reconnaître la phase de la courbe en baignoire.
\end{exemple}

\subsection{Rangs médians (petit échantillon)}

\begin{methode}[-- Fonction de défaillance par les rangs médians]
Pour un échantillon de $N$ durées de vie classées par ordre croissant (rang $i$), on estime :
\[
F(t_i) \approx \frac{i - 0{,}3}{N + 0{,}4} \qquad \text{(approximation de Bénard)}
\]
puis $R(t_i) = 1 - F(t_i)$.
\end{methode}

\begin{appliindus}[-- Durée de vie de roulements]
Les fabricants notent $L_{10}$ la durée de vie nominale telle que $90\,\%$ des roulements l'atteignent, soit $R(L_{10}) = 0{,}90$. C'est un seuil de fiabilité contractuel utilisé pour le dimensionnement des paliers et la planification du remplacement préventif.
\end{appliindus}

\section{Loi exponentielle \hfill {\normalsize\color{uimmred}[C03]\,[C04]}}

\begin{definition}
Lorsque le taux de défaillance est \textbf{constant} ($\lambda(t) = \lambda$), la durée de vie suit une \textbf{loi exponentielle} :
\[
\boxed{R(t) = e^{-\lambda t}}, \qquad F(t) = 1 - e^{-\lambda t}, \qquad f(t) = \lambda\,e^{-\lambda t}.
\]
\end{definition}

\begin{theoreme}[-- MTBF de la loi exponentielle]
\[
\text{MTBF} = \int_0^{+\infty} e^{-\lambda t}\,dt = \frac{1}{\lambda}.
\]
\end{theoreme}

\begin{center}
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=5.5cm,
  xlabel={$t$}, ylabel={$R(t)$},
  xmin=0, xmax=6, ymin=0, ymax=1.05,
  grid=both, grid style={uimmlightblue},
  axis lines=left, tick style={uimmgray},
  every axis label/.style={uimmblue}, samples=100, domain=0:6
]
\addplot[uimmblue, very thick] {exp(-0.5*x)};
\draw[uimmred, dashed] (axis cs:0,0.368) -- (axis cs:2,0.368) -- (axis cs:2,0);
\node[uimmred, font=\footnotesize, anchor=west] at (axis cs:2.1,0.40) {$R(\text{MTBF}) \approx 0{,}37$};
\end{axis}
\end{tikzpicture}
\end{center}

\begin{remarque}
La loi exponentielle est \textbf{sans mémoire} : un composant ``usagé'' a la même probabilité de survie qu'un composant neuf. Elle modélise la \textbf{vie utile} (fond de la baignoire). À $t = \text{MTBF}$, la fiabilité vaut toujours $e^{-1} \approx 0{,}37$.
\end{remarque}

\begin{exemple}[-- Redondance de chariots]
Deux chariots en \textbf{redondance active} suivent $R(t) = e^{-\lambda t}$ avec MTBF $= 54$\,h, donc $\lambda = 1/54$. La fiabilité d'\textbf{un} chariot à $16$\,h vaut $e^{-16/54} \approx 0{,}745$. (Le système parallèle est traité en section 5.)
\end{exemple}

\section{Loi de Weibull \hfill {\normalsize\color{uimmred}[C03]\,[C04]}}

\begin{definition}
La \textbf{loi de Weibull} décrit une grande variété de durées de vie. Avec les paramètres $\beta$ (forme), $\eta$ (échelle, ou durée de vie caractéristique) et $\gamma$ (position) :
\[
\boxed{R(t) = e^{-\left(\frac{t - \gamma}{\eta}\right)^{\beta}}}, \qquad t \geq \gamma.
\]
\end{definition}

\begin{theoreme}[-- Interprétation du paramètre de forme $\beta$]
Le paramètre $\beta$ indique la phase de vie :
\begin{itemize}
  \item $\beta < 1$ : taux de défaillance \textbf{décroissant} (jeunesse) ;
  \item $\beta = 1$ : taux \textbf{constant} ; on retrouve la loi exponentielle avec $\lambda = 1/\eta$ ;
  \item $\beta > 1$ : taux \textbf{croissant} (vieillesse, usure) ; $\beta \approx 3{,}5$ approche la loi normale.
\end{itemize}
\end{theoreme}

\begin{theoreme}[-- MTBF de la loi de Weibull]
\[
\text{MTBF} = \gamma + \eta\,\Gamma\!\left(1 + \frac{1}{\beta}\right)
\]
où $\Gamma$ est la fonction Gamma, lue dans une table. Quelques valeurs de $A = \Gamma(1 + 1/\beta)$ :
\renewcommand{\arraystretch}{1.3}
\begin{center}
\begin{tabular}{|c|c|c|c|c|c|c|}
\hline
\rowcolor{uimmblue}
\textcolor{white}{$\beta$} & \textcolor{white}{$1$} & \textcolor{white}{$1{,}5$} & \textcolor{white}{$2$} & \textcolor{white}{$2{,}5$} & \textcolor{white}{$3$} & \textcolor{white}{$3{,}5$} \\
\hline
$A$ & $1$ & $0{,}903$ & $0{,}886$ & $0{,}887$ & $0{,}893$ & $0{,}900$ \\
\hline
\end{tabular}
\end{center}
\end{theoreme}

\subsection{Ajustement par régression linéaire \hfill {\normalsize\color{uimmred}[C03]}}

\begin{methode}[-- Linéarisation de Weibull (cas $\gamma = 0$)]
À partir de $R(t) = e^{-(t/\eta)^{\beta}}$, on prend deux fois le logarithme :
\[
\ln\!\left(\ln\frac{1}{R(t)}\right) = \beta \ln(t) - \beta \ln(\eta).
\]
En posant $Y = \ln\!\left(\ln\frac{1}{R}\right)$ et $X = \ln(t)$, on obtient une \textbf{droite} $Y = \beta X - \beta\ln\eta$. La régression linéaire (ou le papier d'Allan-Plait) donne :
\begin{itemize}
  \item la pente $= \beta$ (paramètre de forme) ;
  \item l'échelle $\eta = e^{-b/\beta}$ à partir de l'ordonnée à l'origine $b$.
\end{itemize}
Un bon alignement des points valide le modèle de Weibull.
\end{methode}

\begin{appliindus}[-- Analyse de Weibull d'un moteur]
À partir des TBF relevés sur un parc de moteurs, on classe les durées, on calcule $F$ par les rangs médians, puis $X = \ln t$ et $Y = \ln(\ln(1/R))$. La régression linéaire fournit $\beta$ et $\eta$ : on en déduit la phase de vie, la MTBF et la fiabilité à un horizon donné. C'est la base de la planification de la maintenance et des stocks de pièces.
\end{appliindus}

\section{Fiabilité des systèmes \hfill {\normalsize\color{uimmred}[C04]}}

\subsection{Système en série}

\begin{theoreme}
Un système est \textbf{en série} si la défaillance d'un seul composant entraîne la panne de l'ensemble. Les composants étant indépendants :
\[
\boxed{R_{S} = R_1 \times R_2 \times \dots \times R_n = \prod_{i=1}^{n} R_i}
\]
La fiabilité d'un système série est \textbf{inférieure} à celle de son plus faible composant.
\end{theoreme}

\begin{exemple}
Quatre composants en série de fiabilités $0{,}98$ ; $0{,}97$ ; $0{,}95$ ; $0{,}99$ :
\[
R_S = 0{,}98 \times 0{,}97 \times 0{,}95 \times 0{,}99 \approx 0{,}894.
\]
\end{exemple}

\subsection{Système en parallèle (redondance)}

\begin{theoreme}
Un système est \textbf{en parallèle} (redondance) s'il fonctionne tant qu'au moins un composant fonctionne. L'ensemble est défaillant seulement si \textbf{tous} le sont :
\[
\boxed{R_{P} = 1 - \prod_{i=1}^{n} (1 - R_i)}
\]
La redondance \textbf{augmente} la fiabilité au-dessus de celle du meilleur composant.
\end{theoreme}

\begin{exemple}
Deux chariots identiques en redondance, chacun de fiabilité $0{,}745$ à $16$\,h :
\[
R_P = 1 - (1 - 0{,}745)^2 = 1 - 0{,}255^2 \approx 0{,}935.
\]
\end{exemple}

\end{center}

\begin{appliindus}[-- Sécurité aéronautique]
Le système de propulsion d'un avion quadrimoteur est étudié de deux façons : si l'avion s'écrase seulement quand les $4$ moteurs tombent en panne, c'est un montage \textbf{parallèle} (très fiable) ; si la perte d'un seul moteur est critique, c'est un montage \textbf{série} (fiabilité réduite). Le même parc de moteurs donne deux fiabilités très différentes selon l'architecture.
\end{appliindus}

\section{Périodicité d'une intervention préventive \hfill {\normalsize\color{uimmred}[C05]}}

\begin{methode}[-- Période de remplacement systématique]
Pour garantir une fiabilité minimale $R_0$ (par exemple $0{,}90$), on cherche la période $T$ telle que $R(T) = R_0$.
\begin{itemize}
  \item \textbf{Loi exponentielle :} $e^{-\lambda T} = R_0 \Rightarrow T = \dfrac{-\ln R_0}{\lambda}$.
  \item \textbf{Loi de Weibull ($\gamma = 0$) :} $e^{-(T/\eta)^{\beta}} = R_0 \Rightarrow T = \eta\,\left(-\ln R_0\right)^{1/\beta}$.
\end{itemize}
On remplace alors le composant tous les $T$ avant qu'il n'atteigne une fiabilité trop basse.
\end{methode}

\begin{exemple}[-- Changement systématique à $90\,\%$]
Un composant exponentiel de MTBF $200$\,h ($\lambda = 1/200$). Pour $R_0 = 0{,}90$ :
\[
T = \frac{-\ln(0{,}90)}{1/200} = 200 \times 0{,}105 \approx 21\ \text{heures.}
\]
On planifie le remplacement toutes les $21$\,heures pour conserver une fiabilité $\geq 90\,\%$.
\end{exemple}

\begin{appliindus}[-- Maintenance préventive systématique]
La détermination de $T$ est la clé d'un plan de \textbf{maintenance préventive systématique} : remplacer trop tôt coûte cher en pièces et en arrêts, trop tard expose à la panne. L'arbitrage se fait souvent en comparant le coût d'une défaillance non planifiée à celui d'un remplacement préventif.
\end{appliindus}

\section{Exercices d'application}

\subsection*{Exercice 1 -- Systèmes série et parallèle \hfill {\normalsize\color{uimmred}[C04]}}
\begin{enumerate}[label=\alph*)]
  \item Quatre composants en série ont pour fiabilités $0{,}96$ ; $0{,}93$ ; $0{,}90$ ; $0{,}99$. Calculer $R_S$.
  \item Quelle fiabilité minimale du dernier composant donnerait $R_S \geq 0{,}88$ ?
  \item Trois composants en parallèle ont pour fiabilités $0{,}80$ ; $0{,}75$ ; $0{,}70$. Calculer $R_P$.
\end{enumerate}

\subsection*{Exercice 2 -- Taux de défaillance empirique \hfill {\normalsize\color{uimmred}[C02]}}
Sur $150$ capteurs neufs, on relève par intervalle de $100$\,h les nombres de défaillants : $12, 10, 5, 4, 7$.
\begin{enumerate}[label=\alph*)]
  \item Dresser le tableau des survivants en début de chaque intervalle.
  \item Calculer le taux de défaillance empirique de chaque intervalle.
  \item Le taux est-il plutôt décroissant, constant ou croissant ?
\end{enumerate}

\subsection*{Exercice 3 -- Loi exponentielle \hfill {\normalsize\color{uimmred}[C04]}}
Un composant suit $R(t) = e^{-\lambda t}$ avec une MTBF de $400$\,h.
\begin{enumerate}[label=\alph*)]
  \item Déterminer $\lambda$.
  \item Calculer la fiabilité à $100$\,h.
  \item Calculer la probabilité de panne avant $50$\,h.
\end{enumerate}

\subsection*{Exercice 4 -- Vérification d'une loi \hfill {\normalsize\color{uimmred}[C03]}}
On relève les TBF (jours) de $7$ pannes successives : $24, 27, 30, 33, 37, 43, 52$.
\begin{enumerate}[label=\alph*)]
  \item Calculer $F$ et $R$ par les rangs médians, puis $Y = \ln(1/R)$.
  \item La durée de vie suit-elle une loi exponentielle ? Justifier par un alignement.
\end{enumerate}

\subsection*{Exercice 5 -- Analyse de Weibull \hfill {\normalsize\color{uimmred}[C03]\,[C04]}}
Un moteur est étudié par Weibull ; l'ajustement donne $\beta = 1{,}8$, $\eta = 1500$\,h, $\gamma = 0$.
\begin{enumerate}[label=\alph*)]
  \item À quelle phase de la courbe en baignoire correspond ce $\beta$ ?
  \item Calculer la MTBF (utiliser $\Gamma(1 + 1/1{,}8) \approx 0{,}889$).
  \item Calculer la fiabilité à $500$\,h.
\end{enumerate}

\subsection*{Exercice 6 -- Disponibilité \hfill {\normalsize\color{uimmred}[C01]}}
Un équipement a une MTBF de $250$\,h et une MTTR de $6$\,h.
\begin{enumerate}[label=\alph*)]
  \item Calculer la disponibilité.
  \item Quelle MTTR maximale faut-il viser pour une disponibilité de $99\,\%$ ?
\end{enumerate}

\subsection*{Exercice 7 -- Périodicité de maintenance \hfill {\normalsize\color{uimmred}[C05]}}
Un roulement suit une loi de Weibull $\beta = 2$, $\eta = 3000$\,h, $\gamma = 0$.
\begin{enumerate}[label=\alph*)]
  \item Déterminer la période $T$ de remplacement systématique pour garantir $R \geq 0{,}90$.
  \item Comparer à la MTBF du roulement.
\end{enumerate}

\section*{Formulaire -- À retenir}
\addcontentsline{toc}{section}{Formulaire}

\begin{tcolorbox}[colback=uimmlightblue, colframe=uimmblue, title={\textbf{Formulaire de fiabilité}}, fonttitle=\large\bfseries\sffamily, coltitle=white, boxrule=1.5pt, arc=5pt]

\textbf{Fonctions :} $R(t) = P(T>t)$, \; $F(t) = 1 - R(t)$, \; $f(t) = -R'(t)$, \; $\lambda(t) = \dfrac{f(t)}{R(t)}$.

\vspace{5pt}
\textbf{Moyennes :} $\text{MTBF} = \displaystyle\int_0^{+\infty} R(t)\,dt$, \; $D = \dfrac{\text{MTBF}}{\text{MTBF} + \text{MTTR}}$.

\vspace{5pt}
\textbf{Loi exponentielle :} $R(t) = e^{-\lambda t}$, \; $\text{MTBF} = \dfrac{1}{\lambda}$, \; $\lambda$ constant.

\vspace{5pt}
\textbf{Loi de Weibull :} $R(t) = e^{-\left(\frac{t-\gamma}{\eta}\right)^{\beta}}$, \; $\text{MTBF} = \gamma + \eta\,\Gamma\!\left(1 + \tfrac{1}{\beta}\right)$.

\vspace{3pt}
$\beta < 1$ jeunesse \; ; \; $\beta = 1$ vie utile (exponentielle) \; ; \; $\beta > 1$ vieillesse.

\vspace{5pt}
\textbf{Linéarisation Weibull :} $\ln\!\left(\ln\frac{1}{R}\right) = \beta\ln t - \beta\ln\eta$ \; (droite $Y = \beta X + b$).

\vspace{5pt}
\textbf{Systèmes :} série $R_S = \prod R_i$ \; ; \; parallèle $R_P = 1 - \prod (1 - R_i)$.

\vspace{5pt}
\textbf{Périodicité ($R(T) = R_0$) :} exponentielle $T = \dfrac{-\ln R_0}{\lambda}$ \; ; \; Weibull $T = \eta(-\ln R_0)^{1/\beta}$.
\end{tcolorbox}
