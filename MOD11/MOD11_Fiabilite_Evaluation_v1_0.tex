\section{Vocabulaire et disponibilité \hfill {\normalsize\color{uimmred}[C01]}}
\begin{exo}{Exercice 1}
\begin{enumerate}[label=\alph*)]
  \item Définir la fiabilité $R(t)$ et la relier à la fonction de défaillance $F(t)$.
  \item Une machine a une MTBF de $150$\,h et une MTTR de $5$\,h. Calculer sa disponibilité.
\end{enumerate}
\end{exo}

\section{Systèmes \hfill {\normalsize\color{uimmred}[C04]}}
\begin{exo}{Exercice 2}
Un dispositif comporte trois composants A, B, C en série, de fiabilités $0{,}95$ ; $0{,}92$ ; $0{,}98$.
\begin{enumerate}[label=\alph*)]
  \item Calculer la fiabilité de l'ensemble.
  \item On double le composant B en redondance (deux B identiques en parallèle). Calculer la nouvelle fiabilité du bloc B, puis de l'ensemble.
\end{enumerate}
\end{exo}

\section{Loi exponentielle \hfill {\normalsize\color{uimmred}[C04]}}
\begin{exo}{Exercice 3}
Un composant suit $R(t) = e^{-\lambda t}$ avec une MTBF de $250$\,h.
\begin{enumerate}[label=\alph*)]
  \item Déterminer $\lambda$.
  \item Calculer la fiabilité à $100$\,h.
  \item Calculer la probabilité de défaillance avant $40$\,h.
\end{enumerate}
\end{exo}

\section{Loi de Weibull \hfill {\normalsize\color{uimmred}[C03]\,[C04]}}
\begin{exo}{Exercice 4}
Un organe suit une loi de Weibull $\beta = 2$, $\eta = 1000$\,h, $\gamma = 0$.
\begin{enumerate}[label=\alph*)]
  \item À quelle phase de la courbe en baignoire correspond ce $\beta$ ?
  \item Calculer la fiabilité à $300$\,h.
  \item Calculer la MTBF (on prendra $\Gamma(1{,}5) \approx 0{,}886$).
\end{enumerate}
\end{exo}

\section{Périodicité de maintenance \hfill {\normalsize\color{uimmred}[C05]}}
\begin{exo}{Exercice 5}
Reprendre l'organe de l'exercice 4 ($\beta = 2$, $\eta = 1000$\,h).
\begin{enumerate}[label=\alph*)]
  \item Déterminer la période $T$ de remplacement systématique garantissant $R \geq 0{,}90$.
  \item Comparer $T$ à la MTBF et commenter.
\end{enumerate}
\end{exo}
