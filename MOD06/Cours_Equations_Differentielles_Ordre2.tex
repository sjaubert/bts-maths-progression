\section{Introduction et motivations}

Les équations différentielles linéaires du second ordre sont des outils mathématiques fondamentaux pour modéliser de nombreux phénomènes physiques et industriels. Elles décrivent l'évolution de grandeurs physiques dont la variation dépend de la grandeur elle-même et de sa vitesse de variation.

\begin{remarque}
Dans le monde industriel, on rencontre ces équations dans :
\begin{itemize}
  \item Les \textbf{circuits électriques RLC} (électronique, automatisme)
  \item Les \textbf{systèmes masse-ressort-amortisseur} (vibrations de machines)
  \item La \textbf{régulation industrielle} (asservissement, capteurs)
  \item La \textbf{mécanique des structures} (flexion de poutres, oscillations)
  \item La \textbf{thermique} (transferts de chaleur transitoires)
\end{itemize}
\end{remarque}

\subsection{Prérequis}
\begin{itemize}
  \item Équations différentielles linéaires du premier ordre à coefficients constants
  \item Résolution d'équations du second degré (discriminant)
  \item Nombres complexes (forme trigonométrique)
  \item Fonctions exponentielles, trigonométriques
  \item Dérivation et primitives
\end{itemize}

\subsection{Objectifs du module}
À l'issue de ce cours, l'étudiant sera capable de :
\begin{enumerate}
  \item Identifier une équation différentielle linéaire du second ordre à coefficients constants
  \item Résoudre l'équation homogène associée (3 cas du discriminant)
  \item Déterminer une solution particulière
  \item Appliquer des conditions initiales pour trouver la solution unique
  \item Interpréter physiquement la solution (régime transitoire / permanent)
\end{enumerate}

\section{Généralités sur les équations différentielles du 2\textsuperscript{nd} ordre}

\begin{definition}
Une \textbf{équation différentielle linéaire du second ordre à coefficients constants} est une équation de la forme :
\[
\boxed{a\,y''(t) + b\,y'(t) + c\,y(t) = f(t)}
\]
où :
\begin{itemize}
  \item $a$, $b$, $c$ sont des constantes réelles avec $a \neq 0$
  \item $y(t)$ est la fonction inconnue
  \item $y'(t)$ et $y''(t)$ sont ses dérivées première et seconde
  \item $f(t)$ est le \textbf{second membre} (fonction connue)
\end{itemize}
\end{definition}

\begin{definition}
Lorsque $f(t) = 0$, l'équation est dite \textbf{homogène} (ou \textbf{sans second membre}) :
\[
a\,y''(t) + b\,y'(t) + c\,y(t) = 0
\]
On l'appelle aussi \textbf{équation homogène associée} lorsqu'on étudie l'équation complète.
\end{definition}

\begin{exemple}[-- Vocabulaire]
\renewcommand{\arraystretch}{1.4}
\begin{center}
\begin{tabular}{>{\raggedright}p{7cm} c c}
\toprule
\textbf{Équation} & \textbf{Ordre} & \textbf{Type} \\
\midrule
$y'' + 4y' + 3y = 0$ & 2 & Homogène \\
$2y'' + 5y' + y = \sin(t)$ & 2 & Avec second membre \\
$y'' - 9y = e^{2t}$ & 2 & Avec second membre \\
$y'' + \omega^2 y = 0$ & 2 & Homogène \\
\bottomrule
\end{tabular}
\end{center}
\end{exemple}

\begin{appliindus}[-- Mise en équation d'un circuit RLC série]
Considérons un circuit RLC série soumis à une tension $E$ constante. La loi des mailles donne :
\[
L\,\frac{d^2q}{dt^2} + R\,\frac{dq}{dt} + \frac{1}{C}\,q = E
\]
où $q(t)$ est la charge du condensateur, $L$ l'inductance, $R$ la résistance et $C$ la capacité.

En posant $i(t) = q'(t)$ (intensité), on obtient aussi :
\[
L\,\frac{di}{dt} + R\,i + \frac{1}{C}\int_0^t i(\tau)\,d\tau = E
\]
\textbf{Identification :} $a = L$, $b = R$, $c = \frac{1}{C}$, $f(t) = E$.
\end{appliindus}

\begin{appliindus}[-- Système masse-ressort-amortisseur]
Un solide de masse $m$ est attaché à un ressort de raideur $k$ et amorti par un fluide de coefficient de frottement $f$. L'application du PFD (Principe Fondamental de la Dynamique) donne :
\[
m\,\ddot{x}(t) + f\,\dot{x}(t) + k\,x(t) = F_{\text{ext}}(t)
\]
où $x(t)$ est le déplacement par rapport à la position d'équilibre.

\textbf{Identification :} $a = m$, $b = f$, $c = k$, second membre $= F_{\text{ext}}(t)$.
\end{appliindus}

\section{Résolution de l'équation homogène}

\subsection{Équation caractéristique}

\begin{theoreme}
Pour résoudre l'équation homogène $a\,y'' + b\,y' + c\,y = 0$, on recherche des solutions de la forme $y(t) = e^{rt}$.

En substituant dans l'équation, on obtient l'\textbf{équation caractéristique} :
\[
\boxed{a\,r^2 + b\,r + c = 0}
\]
Le discriminant $\Delta = b^2 - 4ac$ détermine la nature des solutions.
\end{theoreme}

\begin{methode}[-- Résolution de l'équation caractéristique]
\begin{enumerate}
  \item Écrire l'équation caractéristique $ar^2 + br + c = 0$
  \item Calculer le discriminant $\Delta = b^2 - 4ac$
  \item Selon le signe de $\Delta$, déterminer les racines et la solution générale
\end{enumerate}
\end{methode}

\subsection{Cas 1 : $\Delta > 0$ -- Deux racines réelles distinctes}

\begin{theoreme}[-- Régime apériodique (sur-amorti)]
Si $\Delta > 0$, l'équation caractéristique admet deux racines réelles distinctes :
\[
r_1 = \frac{-b - \sqrt{\Delta}}{2a} \quad \text{et} \quad r_2 = \frac{-b + \sqrt{\Delta}}{2a}
\]
La \textbf{solution générale} de l'équation homogène est :
\[
\boxed{y_h(t) = A\,e^{r_1 t} + B\,e^{r_2 t}}
\]
où $A$ et $B$ sont des constantes déterminées par les conditions initiales.
\end{theoreme}

\begin{exemple}[1 -- Résolution complète]
Résoudre : $y'' + 5y' + 6y = 0$

\textbf{Équation caractéristique :} $r^2 + 5r + 6 = 0$

$\Delta = 25 - 24 = 1 > 0$

$r_1 = \dfrac{-5 - 1}{2} = -3$ \quad et \quad $r_2 = \dfrac{-5 + 1}{2} = -2$

\textbf{Solution générale :} $y_h(t) = A\,e^{-3t} + B\,e^{-2t}$
\end{exemple}

\begin{appliindus}[-- Décharge d'un circuit RLC sur-amorti]
Circuit avec $L = 1\,$H, $R = 5\,\Omega$, $C = \frac{1}{6}\,$F.

Équation : $q'' + 5q' + 6q = 0$ avec $q(0) = Q_0$ et $q'(0) = i(0) = 0$.

La solution est : $q(t) = 3Q_0\,e^{-2t} - 2Q_0\,e^{-3t}$

\textbf{Interprétation :} Le condensateur se décharge sans oscillation. C'est le \textbf{régime apériodique}. Ce régime est recherché dans les instruments de mesure pour éviter les oscillations parasites.
\end{appliindus}

\subsection{Cas 2 : $\Delta = 0$ -- Racine double}

\begin{theoreme}[-- Régime critique]
Si $\Delta = 0$, l'équation caractéristique admet une racine double :
\[
r_0 = \frac{-b}{2a}
\]
La \textbf{solution générale} est :
\[
\boxed{y_h(t) = (A + B\,t)\,e^{r_0 t}}
\]
\end{theoreme}

\begin{exemple}[2 -- Racine double]
Résoudre : $y'' + 4y' + 4y = 0$

\textbf{Équation caractéristique :} $r^2 + 4r + 4 = 0 \Leftrightarrow (r+2)^2 = 0$

$r_0 = -2$ (racine double)

\textbf{Solution générale :} $y_h(t) = (A + Bt)\,e^{-2t}$
\end{exemple}

\begin{appliindus}[-- Amortissement critique d'une suspension]
La suspension d'un véhicule est modélisée par $m\ddot{x} + f\dot{x} + kx = 0$.

Le \textbf{régime critique} ($\Delta = 0$, soit $f^2 = 4mk$) correspond au retour le plus rapide à l'équilibre \textbf{sans oscillation}. C'est le réglage optimal des amortisseurs automobile.

Si $f_c = 2\sqrt{mk}$, alors $r_0 = -\frac{f_c}{2m}$ et :
\[
x(t) = (x_0 + (v_0 + r_0 x_0)\,t)\,e^{r_0 t}
\]
\end{appliindus}

\subsection{Cas 3 : $\Delta < 0$ -- Racines complexes conjuguées}

\begin{theoreme}[-- Régime pseudo-périodique (sous-amorti)]
Si $\Delta < 0$, l'équation caractéristique admet deux racines complexes conjuguées :
\[
r_1 = \alpha + j\beta \quad \text{et} \quad r_2 = \alpha - j\beta
\]
avec $\alpha = \dfrac{-b}{2a}$ et $\beta = \dfrac{\sqrt{|\Delta|}}{2a}$

La \textbf{solution générale} est :
\[
\boxed{y_h(t) = e^{\alpha t}\left(A\cos(\beta t) + B\sin(\beta t)\right)}
\]
ou de façon équivalente : $y_h(t) = C\,e^{\alpha t}\cos(\beta t + \varphi)$
\end{theoreme}

\begin{remarque}
\begin{itemize}
  \item $\alpha$ est le \textbf{coefficient d'amortissement} : si $\alpha < 0$, les oscillations sont amorties
  \item $\beta$ est la \textbf{pseudo-pulsation} des oscillations
  \item La \textbf{pseudo-période} vaut $T = \frac{2\pi}{\beta}$
\end{itemize}
\end{remarque}

\begin{exemple}[3 -- Racines complexes]
Résoudre : $y'' + 2y' + 5y = 0$

\textbf{Équation caractéristique :} $r^2 + 2r + 5 = 0$

$\Delta = 4 - 20 = -16 < 0$

$\alpha = \dfrac{-2}{2} = -1$, \quad $\beta = \dfrac{\sqrt{16}}{2} = 2$

\textbf{Solution générale :} $y_h(t) = e^{-t}\left(A\cos(2t) + B\sin(2t)\right)$

Les oscillations sont amorties (car $\alpha = -1 < 0$) avec une pseudo-pulsation $\beta = 2\,$rad/s.
\end{exemple}

\begin{appliindus}[-- Oscillations d'un circuit RLC sous-amorti]
Un circuit RLC avec $L = 0{,}1\,$H, $R = 20\,\Omega$, $C = 100\,\mu$F :

$\Delta = R^2 - 4\frac{L}{C} = 400 - 4000 = -3600 < 0$

On obtient des oscillations \textbf{amorties} de la tension aux bornes du condensateur :
\[
u_C(t) = U_0\,e^{-\frac{R}{2L}t}\cos\!\left(\sqrt{\frac{1}{LC} - \frac{R^2}{4L^2}}\;t + \varphi\right)
\]

Ce phénomène est utilisé dans les \textbf{oscillateurs électroniques} et les \textbf{filtres passe-bande}.
\end{appliindus}

\section{Équation complète avec second membre}

\begin{theoreme}[-- Structure de la solution générale]
La solution générale de l'équation complète $a\,y'' + b\,y' + c\,y = f(t)$ est :
\[
\boxed{y(t) = y_h(t) + y_p(t)}
\]
où :
\begin{itemize}
  \item $y_h(t)$ est la solution générale de l'équation homogène (\textbf{régime transitoire})
  \item $y_p(t)$ est une solution particulière de l'équation complète (\textbf{régime permanent})
\end{itemize}
\end{theoreme}

\subsection{Recherche de solution particulière -- Second membre constant}

\begin{methode}[-- $f(t) = K$ (constante)]
Si $c \neq 0$, on cherche $y_p$ constante : $y_p = \dfrac{K}{c}$.
\end{methode}

\begin{exemple}[4]
$y'' + 3y' + 2y = 10$

Solution particulière : $y_p = \dfrac{10}{2} = 5$

Équation caractéristique : $r^2 + 3r + 2 = 0 \Rightarrow r_1 = -1,\; r_2 = -2$

\textbf{Solution générale :} $y(t) = A\,e^{-t} + B\,e^{-2t} + 5$
\end{exemple}

\subsection{Recherche de solution particulière -- Second membre polynomial}

\begin{methode}[-- $f(t) = $ polynôme de degré $n$]
On cherche $y_p$ sous forme d'un polynôme de même degré $n$ (si $c \neq 0$).
\end{methode}

\begin{exemple}[5]
$y'' + y' + y = 3t + 1$

On pose $y_p = \alpha t + \beta$. Alors $y_p' = \alpha$ et $y_p'' = 0$.

Substitution : $0 + \alpha + \alpha t + \beta = 3t + 1$

Par identification : $\alpha = 3$ et $\alpha + \beta = 1 \Rightarrow \beta = -2$

\textbf{Solution particulière :} $y_p(t) = 3t - 2$
\end{exemple}

\subsection{Recherche de solution particulière -- Second membre exponentiel}

\begin{methode}[-- $f(t) = P\,e^{\gamma t}$]
\begin{itemize}
  \item Si $\gamma$ n'est \textbf{pas} racine de l'équation caractéristique : $y_p = \lambda\,e^{\gamma t}$
  \item Si $\gamma$ est racine \textbf{simple} : $y_p = \lambda\,t\,e^{\gamma t}$
  \item Si $\gamma$ est racine \textbf{double} : $y_p = \lambda\,t^2\,e^{\gamma t}$
\end{itemize}
\end{methode}

\begin{exemple}[6]
$y'' - 3y' + 2y = 4e^{3t}$

Équation caractéristique : $r^2 - 3r + 2 = 0 \Rightarrow r_1 = 1,\; r_2 = 2$

$\gamma = 3$ n'est pas racine. On pose $y_p = \lambda\,e^{3t}$.

$y_p'' - 3y_p' + 2y_p = 9\lambda e^{3t} - 9\lambda e^{3t} + 2\lambda e^{3t} = 2\lambda e^{3t} = 4e^{3t}$

Donc $\lambda = 2$ et $y_p = 2e^{3t}$.

\textbf{Solution générale :} $y(t) = A\,e^{t} + B\,e^{2t} + 2e^{3t}$
\end{exemple}

\subsection{Recherche de solution particulière -- Second membre sinusoïdal}

\begin{methode}[-- $f(t) = P\cos(\omega t) + Q\sin(\omega t)$]
On cherche $y_p$ sous la forme : $y_p(t) = \lambda\cos(\omega t) + \mu\sin(\omega t)$

On substitue et on identifie les coefficients de $\cos(\omega t)$ et $\sin(\omega t)$.
\end{methode}

\begin{exemple}[7]
$y'' + 4y = \cos(t)$

On pose $y_p = \lambda\cos(t) + \mu\sin(t)$.

$y_p'' = -\lambda\cos(t) - \mu\sin(t)$

$y_p'' + 4y_p = 3\lambda\cos(t) + 3\mu\sin(t) = \cos(t)$

Donc $\lambda = \frac{1}{3}$, $\mu = 0$.

\textbf{Solution particulière :} $y_p(t) = \dfrac{1}{3}\cos(t)$
\end{exemple}

\begin{appliindus}[-- Vibrations forcées d'une machine-outil]
Une machine de masse $m = 500\,$kg, montée sur des ressorts ($k = 20\,000\,$N/m) avec amortisseur ($f = 1\,000\,$N$\cdot$s/m), soumise à une force excitatrice $F(t) = 200\sin(10t)$ :
\[
500\,\ddot{x} + 1000\,\dot{x} + 20000\,x = 200\sin(10t)
\]
Simplifiée : $\ddot{x} + 2\dot{x} + 40x = 0{,}4\sin(10t)$

La solution particulière donne l'amplitude et le déphasage des vibrations en régime permanent. Ce calcul est essentiel pour le \textbf{dimensionnement des plots anti-vibratoires} en milieu industriel.
\end{appliindus}

\section{Détermination des constantes par conditions initiales}

\begin{theoreme}[-- Existence et unicité]
L'équation $a\,y'' + b\,y' + c\,y = f(t)$ admet une \textbf{unique solution} vérifiant les conditions initiales :
\[
y(t_0) = y_0 \quad \text{et} \quad y'(t_0) = y_0'
\]
Ces deux conditions permettent de déterminer les constantes $A$ et $B$.
\end{theoreme}

\begin{methode}[-- Détermination de $A$ et $B$]
\begin{enumerate}
  \item Écrire la solution générale $y(t) = y_h(t) + y_p(t)$
  \item Calculer $y'(t)$
  \item Appliquer $y(t_0) = y_0$ : obtenir une première équation en $A$ et $B$
  \item Appliquer $y'(t_0) = y_0'$ : obtenir une seconde équation en $A$ et $B$
  \item Résoudre le système de 2 équations à 2 inconnues
\end{enumerate}
\end{methode}

\begin{exemple}[8 -- Problème de Cauchy complet]
Résoudre : $y'' + 2y' + 5y = 5$ avec $y(0) = 0$ et $y'(0) = 0$.

\textbf{1) Solution particulière :} $y_p = \frac{5}{5} = 1$

\textbf{2) Équation homogène :} $r^2 + 2r + 5 = 0$, $\Delta = 4 - 20 = -16 < 0$

$\alpha = -1$, $\beta = 2$

$y_h(t) = e^{-t}(A\cos(2t) + B\sin(2t))$

\textbf{3) Solution générale :} $y(t) = e^{-t}(A\cos(2t) + B\sin(2t)) + 1$

\textbf{4) Conditions initiales :}

$y(0) = A + 1 = 0 \Rightarrow A = -1$

$y'(t) = e^{-t}[(-A+2B)\cos(2t) + (-B-2A)\sin(2t)]$

$y'(0) = -A + 2B = 1 + 2B = 0 \Rightarrow B = -\frac{1}{2}$

\textbf{Solution :} $\displaystyle y(t) = 1 - e^{-t}\!\left(\cos(2t) + \frac{1}{2}\sin(2t)\right)$
\end{exemple}

\begin{appliindus}[-- Réponse indicielle d'un capteur de température]
Un capteur de température industriel (thermocouple dans un doigt de gant) peut être modélisé par un système du second ordre. Lorsqu'il est plongé dans un fluide à $100\,°$C à $t=0$ :
\[
\tau^2\,\ddot{T} + 2m\tau\,\dot{T} + T = 100
\]
avec $T(0) = 20\,°$C (température initiale) et $T'(0) = 0$.

Le facteur d'amortissement $m$ détermine la rapidité et la qualité de la réponse du capteur :
\begin{itemize}
  \item $m > 1$ : réponse lente sans dépassement
  \item $m = 1$ : réponse la plus rapide sans dépassement
  \item $m < 1$ : réponse rapide mais avec des oscillations
\end{itemize}

Le choix du capteur dans un \textbf{process industriel} dépend de ce compromis rapidité/précision.
\end{appliindus}

L'équation du circuit s'écrit :
\[
L\,\frac{d^2u_C}{dt^2} + R\,\frac{du_C}{dt} + \frac{u_C}{1} = E
\]
En divisant par $LC$ et en posant $\omega_0 = \frac{1}{\sqrt{LC}}$ (pulsation propre) et $Q = \frac{1}{R}\sqrt{\frac{L}{C}}$ (facteur de qualité) :
\[
\frac{d^2u_C}{dt^2} + \frac{\omega_0}{Q}\frac{du_C}{dt} + \omega_0^2\,u_C = \omega_0^2\,E
\]

\begin{remarque}
La nature du régime transitoire dépend du facteur de qualité $Q$ :
\begin{itemize}
  \item $Q < \frac{1}{2}$ $\Leftrightarrow$ $\Delta > 0$ : régime \textbf{apériodique}
  \item $Q = \frac{1}{2}$ $\Leftrightarrow$ $\Delta = 0$ : régime \textbf{critique}
  \item $Q > \frac{1}{2}$ $\Leftrightarrow$ $\Delta < 0$ : régime \textbf{pseudo-périodique}
\end{itemize}
\end{remarque}

\subsection{Système masse-ressort-amortisseur}

Le mouvement d'une masse $m$ suspendue à un ressort de raideur $k$, avec frottement visqueux $f$, est régi par :
\[
m\,\ddot{x} + f\,\dot{x} + k\,x = F(t)
\]

On pose la pulsation propre $\omega_0 = \sqrt{\frac{k}{m}}$ et le coefficient d'amortissement réduit $m = \frac{f}{2\sqrt{mk}}$ :
\[
\ddot{x} + 2m\omega_0\,\dot{x} + \omega_0^2\,x = \frac{F(t)}{m}
\]

\begin{appliindus}[-- Dimensionnement d'un amortisseur de machine]
Une presse industrielle de $m = 200\,$kg montée sur plots élastiques ($k = 8\,000\,$N/m). On souhaite un retour à l'équilibre \textbf{critique} ($m = 1$) :

$f_c = 2\sqrt{mk} = 2\sqrt{200 \times 8000} = 2\sqrt{1\,600\,000} \approx 2\,530\,$N$\cdot$s/m

L'amortisseur doit avoir un coefficient de frottement visqueux de $\approx 2\,530\,$N$\cdot$s/m.
\end{appliindus}

\subsection{Oscillations de torsion}

\begin{appliindus}[-- Arbre de transmission]
Un arbre de transmission subit des oscillations de torsion. En notant $\theta(t)$ l'angle de torsion, $J$ le moment d'inertie, $c$ le coefficient de raideur de torsion et $\mu$ le coefficient d'amortissement :
\[
J\,\ddot{\theta} + \mu\,\dot{\theta} + c\,\theta = \Gamma_{\text{ext}}(t)
\]
La détermination du régime permet de prédire les vibrations et d'éviter les phénomènes de \textbf{résonance} qui peuvent endommager l'arbre.
\end{appliindus}

\subsection{Régulation et automatisme}

\begin{appliindus}[-- Système asservi du second ordre]
En régulation industrielle, de nombreux systèmes (positionnement de bras robotique, régulation de débit, contrôle de niveau) sont modélisés par un système du second ordre. La fonction de transfert a la forme canonique :
\[
H(p) = \frac{K}{1 + \frac{2m}{\omega_0}p + \frac{p^2}{\omega_0^2}}
\]
La réponse temporelle à un échelon est la solution de l'équation différentielle associée. Les critères de performance industrielle sont :
\begin{itemize}
  \item \textbf{Temps de réponse à 5\%} : $t_r \approx \frac{3}{m\omega_0}$
  \item \textbf{Dépassement} : $D = e^{-\frac{\pi m}{\sqrt{1-m^2}}} \times 100\,\%$
  \item \textbf{Erreur statique} : dépend du gain $K$
\end{itemize}
\end{appliindus}

\section{Exercices d'application}

\subsection*{Exercice 1 -- Résolution d'équation homogène}
Résoudre les équations différentielles suivantes :
\begin{enumerate}[label=\alph*)]
  \item $y'' + 7y' + 12y = 0$
  \item $y'' - 6y' + 9y = 0$
  \item $y'' + 2y' + 10y = 0$
  \item $y'' - y = 0$
\end{enumerate}

\subsection*{Exercice 2 -- Avec second membre}
Résoudre les équations suivantes :
\begin{enumerate}[label=\alph*)]
  \item $y'' + 4y' + 3y = 6$
  \item $y'' - 2y' + y = 2e^{3t}$
  \item $y'' + 9y = \cos(2t)$
  \item $y'' + y' - 2y = 4t$
\end{enumerate}

\subsection*{Exercice 3 -- Problème de Cauchy}
Résoudre $y'' + 6y' + 8y = 16$ avec $y(0) = 0$ et $y'(0) = 2$.

\subsection*{Exercice 4 -- Circuit RLC}
Un circuit RLC série est constitué d'une résistance $R = 100\,\Omega$, d'une inductance $L = 0{,}5\,$H et d'un condensateur $C = 50\,\mu$F. À $t=0$, on ferme l'interrupteur sur une source de tension $E = 12\,$V.
\begin{enumerate}[label=\alph*)]
  \item Écrire l'équation différentielle vérifiée par $u_C(t)$.
  \item Déterminer le régime de fonctionnement (apériodique, critique ou pseudo-périodique).
  \item Résoudre avec $u_C(0) = 0$ et $u_C'(0) = 0$.
  \item Tracer l'allure de $u_C(t)$.
\end{enumerate}

\subsection*{Exercice 5 -- Vibrations de machine}
Un moteur électrique de masse $m = 150\,$kg est fixé sur des silentblocs de raideur totale $k = 60\,000\,$N/m. Le coefficient d'amortissement est $f = 3\,000\,$N$\cdot$s/m.
\begin{enumerate}[label=\alph*)]
  \item Calculer la pulsation propre $\omega_0$ et le coefficient d'amortissement $m$.
  \item Déterminer le type de régime.
  \item Si le moteur est écarté de $5\,$mm de sa position d'équilibre puis relâché sans vitesse initiale, déterminer $x(t)$.
  \item Calculer la pseudo-période des oscillations et le temps de réponse à 5\%.
\end{enumerate}

\subsection*{Exercice 6 -- Pendule amorti}
Un pendule simple de longueur $\ell = 1\,$m, amorti par frottement visqueux ($\mu = 0{,}5\,$N$\cdot$s/m pour une masse de $0{,}2\,$kg), vérifie (petits angles) :
\[
\ddot{\theta} + \frac{\mu}{m}\,\dot{\theta} + \frac{g}{\ell}\,\theta = 0
\]
\begin{enumerate}[label=\alph*)]
  \item Calculer les paramètres $\omega_0$ et $m$.
  \item Résoudre avec $\theta(0) = 0{,}1\,$rad et $\dot{\theta}(0) = 0$.
  \item Au bout de combien de périodes l'amplitude a-t-elle diminué de moitié ?
\end{enumerate}

\textbf{Forme générale :} $a\,y'' + b\,y' + c\,y = f(t)$ avec $a \neq 0$

\vspace{6pt}
\textbf{Équation caractéristique :} $ar^2 + br + c = 0$, \quad $\Delta = b^2 - 4ac$

\vspace{6pt}
\renewcommand{\arraystretch}{1.8}
\begin{tabular}{|c|c|}
\hline
\rowcolor{uimmblue!20}
\textbf{Cas} & \textbf{Solution homogène $y_h(t)$} \\
\hline
$\Delta > 0$ : $r_1, r_2 \in \mathbb{R}$ & $A\,e^{r_1 t} + B\,e^{r_2 t}$ \\
\hline
$\Delta = 0$ : $r_0$ double & $(A + Bt)\,e^{r_0 t}$ \\
\hline
$\Delta < 0$ : $\alpha \pm j\beta$ & $e^{\alpha t}(A\cos\beta t + B\sin\beta t)$ \\
\hline
\end{tabular}

\vspace{8pt}
\textbf{Solution générale :} $y(t) = y_h(t) + y_p(t)$

\vspace{6pt}
\textbf{Paramètres physiques :}

$\omega_0 = \sqrt{\dfrac{c}{a}}$ \quad (pulsation propre) \qquad
$m = \dfrac{b}{2\sqrt{ac}}$ \quad (amortissement réduit)

\vspace{6pt}
$T_0 = \dfrac{2\pi}{\omega_0}$ \quad (période propre) \qquad
$T_{ps} = \dfrac{2\pi}{\omega_0\sqrt{1-m^2}}$ \quad (pseudo-période, $m < 1$)
\end{tcolorbox}
