\section{Fonctions quadratiques \hfill {\normalsize\color{uimmred}[S01]\,[C01]}}

\begin{exo}{Exercice 1 -- Forme canonique et sommet}
Soit $f(x) = x^2 - 6x + 5$.
\begin{enumerate}[label=\alph*)]
  \item Déterminer les coordonnées du sommet.
  \item Donner la forme canonique.
  \item Déterminer les racines et la forme factorisée.
  \item Dresser le tableau de signes de $f$.
\end{enumerate}
\end{exo}

\begin{exo}{Exercice 2 -- Parabole tournée vers le bas}
Soit $g(x) = -2x^2 + 12x - 10$.
\begin{enumerate}[label=\alph*)]
  \item Déterminer le sommet et le sens de variation.
  \item Pour quelle valeur de $x$ la fonction est-elle maximale ? Que vaut ce maximum ?
\end{enumerate}
\end{exo}

\begin{exo}{Exercice 3 -- Optimisation}
On dispose de $40\,$m de clôture pour délimiter un enclos rectangulaire contre un mur (un seul côté est le mur, donc trois côtés à clôturer). On note $x$ la largeur.
\begin{enumerate}[label=\alph*)]
  \item Exprimer l'aire $A(x)$ de l'enclos.
  \item Déterminer la largeur qui maximise l'aire, puis l'aire maximale.
\end{enumerate}
\end{exo}

\section{Fractions rationnelles \hfill {\normalsize\color{uimmred}[C02]}}

\begin{exo}{Exercice 4 -- Domaine et signe}
Soit $f(x) = \dfrac{2x - 4}{x - 3}$.
\begin{enumerate}[label=\alph*)]
  \item Déterminer le domaine de définition.
  \item Résoudre $f(x) = 0$.
  \item Étudier le signe de $f(x)$.
\end{enumerate}
\end{exo}

\begin{exo}{Exercice 5 -- Équations et inéquations}
\begin{enumerate}[label=\alph*)]
  \item Résoudre $\dfrac{x + 1}{x - 2} = 3$.
  \item Résoudre l'inéquation $\dfrac{x - 5}{x + 1} \leq 0$.
\end{enumerate}
\end{exo}

\section{Logarithme et exponentielle \hfill {\normalsize\color{uimmred}[C03]}}

\begin{exo}{Exercice 6 -- Équations}
Résoudre dans $\mathbb{R}$.
\begin{enumerate}[label=\alph*)]
  \item $e^{2x} = 9$
  \item $\ln(3x - 1) = 2$
  \item $e^{2x} - 4e^x + 3 = 0$
  \item $\ln(x) + \ln(x - 2) = \ln(3)$
\end{enumerate}
\end{exo}

\begin{exo}{Exercice 7 -- Propriétés algébriques}
Simplifier les expressions suivantes.
\begin{enumerate}[label=\alph*)]
  \item $\ln(8) - \ln(2)$
  \item $\ln(e^3) + e^{\ln 5}$
  \item $\ln\!\left(\dfrac{1}{e^2}\right)$
\end{enumerate}
\end{exo}

\begin{exo}{Exercice 8 -- Charge d'un condensateur}
La tension d'un condensateur en charge vaut $u(t) = 10\,(1 - e^{-t/0{,}2})$ (V, $t$ en s).
\begin{enumerate}[label=\alph*)]
  \item Calculer $u(0)$ et la limite de $u$ en $+\infty$.
  \item À quel instant $t$ la tension atteint-elle $8\,$V ?
\end{enumerate}
\end{exo}

\section{Fonctions circulaires \hfill {\normalsize\color{uimmred}[S02]\,[C01]\,[C02]\,[C03]}}

\begin{exo}{Exercice 9 -- Valeurs et équations}
\begin{enumerate}[label=\alph*)]
  \item Donner les valeurs exactes de $\cos\dfrac{\pi}{6}$, $\sin\dfrac{\pi}{4}$, $\tan\dfrac{\pi}{3}$.
  \item Résoudre $\cos x = \dfrac{\sqrt{3}}{2}$ sur $[0\,;\,2\pi[$.
  \item Résoudre $\sin x = -\dfrac{1}{2}$ sur $[0\,;\,2\pi[$.
\end{enumerate}
\end{exo}

\begin{exo}{Exercice 10 -- Fonctions réciproques}
\begin{enumerate}[label=\alph*)]
  \item Calculer $\arcsin\!\left(\dfrac{\sqrt{2}}{2}\right)$, $\arccos(0)$, $\arctan(\sqrt{3})$.
  \item En électricité, le déphasage vaut $\varphi = \arctan\!\left(\dfrac{X}{R}\right)$. Calculer $\varphi$ pour $X = 50\,\Omega$ et $R = 50\,\Omega$.
\end{enumerate}
\end{exo}

\section{Fonctions hyperboliques \hfill {\normalsize\color{uimmred}[S03]\,[C01]\,[C02]\,[C03]}}

\begin{exo}{Exercice 11 -- Définitions et propriétés}
\begin{enumerate}[label=\alph*)]
  \item À partir des définitions, montrer que $\cosh^2 x - \sinh^2 x = 1$.
  \item Calculer $\cosh(0)$, $\sinh(0)$, $\tanh(0)$.
  \item Donner les dérivées de $\sinh x$, $\cosh x$ et $\tanh x$.
\end{enumerate}
\end{exo}

\begin{exo}{Exercice 12 -- Réciproques et chaînette}
\begin{enumerate}[label=\alph*)]
  \item Calculer $\operatorname{argsinh}(0)$ et $\operatorname{argtanh}\!\left(\dfrac{1}{2}\right)$ (valeur exacte puis approchée).
  \item Un câble suspendu a pour équation $y = 5\cosh\!\left(\dfrac{x}{5}\right)$. Calculer la hauteur du câble en $x = 0$ et en $x = 5$ (valeur approchée).
\end{enumerate}
\end{exo}
