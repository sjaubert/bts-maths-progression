\section{Fonction quadratique \hfill {\normalsize\color{uimmred}[C01]}}
\begin{exo}{Exercice 1}
Soit $f(x) = x^2 - 4x - 5$.
\begin{enumerate}[label=\alph*)]
  \item Déterminer le sommet et le sens de variation.
  \item Déterminer les racines.
  \item Donner le signe de $f$.
\end{enumerate}
\end{exo}

\section{Fraction rationnelle \hfill {\normalsize\color{uimmred}[C02]}}
\begin{exo}{Exercice 2}
Soit $g(x) = \dfrac{3x - 6}{x + 2}$.
\begin{enumerate}[label=\alph*)]
  \item Donner le domaine de définition.
  \item Résoudre $g(x) = 0$.
  \item Étudier le signe de $g(x)$.
\end{enumerate}
\end{exo}

\section{Logarithme et exponentielle \hfill {\normalsize\color{uimmred}[C03]}}
\begin{exo}{Exercice 3}
Résoudre dans $\mathbb{R}$.
\begin{enumerate}[label=\alph*)]
  \item $e^{3x} = 4$
  \item $\ln(2x + 1) = 0$
  \item $e^{2x} - 6e^x + 5 = 0$
\end{enumerate}
\end{exo}

\section{Trigonométrie \hfill {\normalsize\color{uimmred}[C01]\,[C02]\,[C03]}}
\begin{exo}{Exercice 4}
\begin{enumerate}[label=\alph*)]
  \item Résoudre $\cos x = \dfrac{1}{2}$ sur $[0\,;\,2\pi[$.
  \item Calculer $\arctan(1)$ et $\arcsin\!\left(-\dfrac{1}{2}\right)$.
\end{enumerate}
\end{exo}

\section{Fonctions hyperboliques \hfill {\normalsize\color{uimmred}[C01]\,[C02]}}
\begin{exo}{Exercice 5}
\begin{enumerate}[label=\alph*)]
  \item Rappeler les définitions de $\cosh x$ et $\sinh x$ à partir de l'exponentielle.
  \item Vérifier que $\cosh^2 x - \sinh^2 x = 1$.
  \item Donner la dérivée de $h(x) = \sinh(2x)$.
\end{enumerate}
\end{exo}
