\begin{consigne}
\textbf{Consignes.} Justifier chaque réponse. Les résultats fractionnaires sont donnés sous forme irréductible. La présentation et la rigueur des calculs sont prises en compte.
\end{consigne}

\section{Logique \hfill {\normalsize\color{uimmred}[C01]}}
\begin{exo}{Exercice 1}
\begin{enumerate}[label=\alph*)]
  \item Indiquer si la proposition $\forall x \in \mathbb{R},\; x^2 > 0$ est vraie ou fausse, et justifier.
  \item Écrire la négation de $\exists n \in \mathbb{N},\; 3n = 21$.
  \item Énoncer la réciproque de : ``si un nombre est divisible par $10$, alors il est pair''. Est-elle vraie ?
\end{enumerate}
\end{exo}

\section{Calcul numérique \hfill {\normalsize\color{uimmred}[C02]\,[C03]}}
\begin{exo}{Exercice 2}
Calculer en respectant les priorités.
\begin{enumerate}[label=\alph*)]
  \item $-6 + 4 \times (-2)^2$
  \item $\dfrac{10^4 \times 10^{-1}}{10^{2}}$
\end{enumerate}
\end{exo}

\begin{exo}{Exercice 3}
Donner sous forme de fraction irréductible.
\begin{enumerate}[label=\alph*)]
  \item $\dfrac{7}{12} - \dfrac{1}{4}$
  \item $\dfrac{4}{9} \div \dfrac{8}{3}$
\end{enumerate}
\end{exo}

\section{Calcul littéral \hfill {\normalsize\color{uimmred}[C04]}}
\begin{exo}{Exercice 4}
\begin{enumerate}[label=\alph*)]
  \item Développer et réduire : $(2x - 3)(x + 5)$.
  \item Développer : $(3x - 1)^2$.
  \item Factoriser : $x^2 - 64$.
  \item Factoriser : $5x^2 + 15x$.
\end{enumerate}
\end{exo}

\section{Équations \hfill {\normalsize\color{uimmred}[C05]\,[C06]}}
\begin{exo}{Exercice 5}
\begin{enumerate}[label=\alph*)]
  \item Résoudre $6x - 5 = 2x + 11$.
  \item Résoudre le système $\begin{cases} x + y = 9 \\ 2x - y = 6 \end{cases}$
  \item Résoudre $x^2 - 8x + 15 = 0$.
\end{enumerate}
\end{exo}

\section{Problème \hfill {\normalsize\color{uimmred}[C07]}}
\begin{exo}{Exercice 6}
Un atelier loue une machine : un forfait de $120\,$€ plus $15\,$€ par jour d'utilisation. Le coût total d'une location est $315\,$€. Poser une équation et déterminer la durée de location.
\end{exo}

\begin{exo}{Exercice 7}
Un cadre rectangulaire a un périmètre de $46\,$cm. Sa longueur dépasse sa largeur de $7\,$cm. Poser le problème et déterminer ses dimensions.
\end{exo}
