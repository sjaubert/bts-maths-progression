\section{Généralités sur les suites \hfill {\normalsize\color{uimmred}[S01]}}

\subsection{Notion de suite}

\begin{definition}
Une \textbf{suite numérique} $(u_n)$ est une fonction qui, à tout entier naturel $n$, associe un nombre réel noté $u_n$ (le \textbf{terme de rang} $n$).
\[
u : \mathbb{N} \rightarrow \mathbb{R}, \qquad n \mapsto u_n
\]
$u_0$ est le \textbf{terme initial}, $u_n$ le terme général, $u_{n+1}$ le terme suivant.
\end{definition}

\begin{remarque}
Ne pas confondre le \textbf{rang} $n$ (la place du terme) et la \textbf{valeur} $u_n$ (le nombre). Par exemple, si $u_n = 2n + 1$, le terme de rang $3$ vaut $u_3 = 7$.
\end{remarque}

\subsection{Deux modes de génération \hfill {\normalsize\color{uimmred}[C01]}}

\begin{definition}[-- Forme explicite]
La suite est définie par une formule donnant directement $u_n$ en fonction de $n$ :
\[
u_n = f(n).
\]
\end{definition}

\begin{definition}[-- Forme récurrente]
La suite est définie par son premier terme et une relation reliant un terme au précédent :
\[
\begin{cases} u_0 \text{ donné} \\ u_{n+1} = f(u_n) \end{cases}
\]
\end{definition}

\begin{exemple}[-- Les deux formes]
\begin{itemize}
  \item Explicite : $u_n = 3n - 1$ donne $u_0 = -1,\; u_1 = 2,\; u_2 = 5,\dots$
  \item Récurrente : $u_0 = 2,\; u_{n+1} = 2u_n + 1$ donne $u_1 = 5,\; u_2 = 11,\; u_3 = 23,\dots$
\end{itemize}
\end{exemple}

\begin{methode}[-- Calculer un terme à la calculatrice ou au tableur]
\begin{itemize}
  \item \textbf{Forme explicite :} on remplace $n$ par sa valeur dans la formule.
  \item \textbf{Forme récurrente au tableur :} en colonne A le rang, en colonne B la valeur. On saisit $u_0$ en B2, puis en B3 la formule traduisant $u_{n+1} = f(u_n)$ (par exemple \texttt{=2*B2+1}), que l'on recopie vers le bas.
\end{itemize}
\end{methode}

\subsection{Calcul des termes par algorithme \hfill {\normalsize\color{uimmred}[C01]}}

Un algorithme permet d'automatiser le calcul des termes d'une suite récurrente.

\begin{lstlisting}[style=uimmpython, caption={Calcul des premiers termes d'une suite récurrente}]
# Suite : u0 = 2 ; u(n+1) = 2*u(n) + 1
u = 2
print("u0 =", u)
for n in range(1, 6):
    u = 2 * u + 1      # nouvelle valeur a partir de l'ancienne
    print("u", n, "=", u)
\end{lstlisting}

\subsection{Représentation graphique \hfill {\normalsize\color{uimmred}[C02]}}

\begin{definition}
On représente une suite par un \textbf{nuage de points} de coordonnées $(n\,;\,u_n)$. Les points ne sont pas reliés : une suite n'est définie que pour les entiers.
\end{definition}

\subsection{Sens de variation}

\begin{definition}
La suite $(u_n)$ est :
\begin{itemize}
  \item \textbf{croissante} si $u_{n+1} \geq u_n$ pour tout $n$ ;
  \item \textbf{décroissante} si $u_{n+1} \leq u_n$ pour tout $n$ ;
  \item \textbf{constante} si $u_{n+1} = u_n$ pour tout $n$.
\end{itemize}
\end{definition}

\begin{methode}[-- Étudier le sens de variation]
On calcule le signe de la différence $u_{n+1} - u_n$ :
\begin{itemize}
  \item si $u_{n+1} - u_n \geq 0$ pour tout $n$ : la suite est croissante ;
  \item si $u_{n+1} - u_n \leq 0$ pour tout $n$ : la suite est décroissante.
\end{itemize}
\end{methode}

\section{Suites arithmétiques \hfill {\normalsize\color{uimmred}[S02]}}

\subsection{Définition}

\begin{definition}
Une suite $(u_n)$ est \textbf{arithmétique} si l'on passe d'un terme au suivant en ajoutant toujours le même nombre $r$, appelé \textbf{raison} :
\[
u_{n+1} = u_n + r.
\]
\end{definition}

\begin{methode}[-- Reconnaître une suite arithmétique]
La différence entre deux termes consécutifs est constante : $u_{n+1} - u_n = r$. Si ce n'est pas le cas, la suite n'est pas arithmétique.
\end{methode}

\subsection{Terme général \hfill {\normalsize\color{uimmred}[C01]}}

\begin{theoreme}
Pour une suite arithmétique de premier terme $u_0$ et de raison $r$ :
\[
\boxed{u_n = u_0 + n\,r}
\]
Plus généralement, à partir du terme de rang $p$ : $u_n = u_p + (n - p)\,r$.
\end{theoreme}

\begin{exemple}
$u_0 = 5$, raison $r = 3$. Alors $u_n = 5 + 3n$, et $u_{10} = 5 + 30 = 35$.
\end{exemple}

\subsection{Somme de termes consécutifs \hfill {\normalsize\color{uimmred}[C02]}}

\begin{theoreme}
La somme de termes consécutifs d'une suite arithmétique vaut :
\[
\boxed{S = (\text{nombre de termes}) \times \frac{\text{premier terme} + \text{dernier terme}}{2}}
\]
En particulier : $\displaystyle \sum_{k=0}^{n} u_k = (n+1)\,\frac{u_0 + u_n}{2}$.
\end{theoreme}

\begin{exemple}
Somme des entiers de $1$ à $100$ : $1 + 2 + \dots + 100 = 100 \times \dfrac{1 + 100}{2} = 5050$.
\end{exemple}

\begin{appliindus}[-- Empilement et cadence linéaire]
Une ligne de production monte en cadence : $20$ pièces la première heure, puis $5$ de plus à chaque heure. La production horaire $u_n = 20 + 5n$ est arithmétique. La production cumulée sur une journée de $8$\,h se calcule avec la formule de la somme, utile pour le suivi de capacité.
\end{appliindus}

\section{Suites géométriques \hfill {\normalsize\color{uimmred}[S02]}}

\subsection{Définition}

\begin{definition}
Une suite $(u_n)$ est \textbf{géométrique} si l'on passe d'un terme au suivant en multipliant toujours par le même nombre $q$ non nul, appelé \textbf{raison} :
\[
u_{n+1} = q \times u_n.
\]
\end{definition}

\begin{methode}[-- Reconnaître une suite géométrique]
Le quotient de deux termes consécutifs est constant : $\dfrac{u_{n+1}}{u_n} = q$ (avec $u_n \neq 0$).
\end{methode}

\subsection{Terme général \hfill {\normalsize\color{uimmred}[C01]}}

\begin{theoreme}
Pour une suite géométrique de premier terme $u_0$ et de raison $q$ :
\[
\boxed{u_n = u_0 \times q^{\,n}}
\]
Plus généralement : $u_n = u_p \times q^{\,n-p}$.
\end{theoreme}

\begin{exemple}
$u_0 = 1000$, raison $q = 1{,}05$. Alors $u_n = 1000 \times 1{,}05^{\,n}$, et $u_4 = 1000 \times 1{,}05^4 \approx 1215{,}5$.
\end{exemple}

\subsection{Somme de termes consécutifs \hfill {\normalsize\color{uimmred}[C02]}}

\begin{theoreme}
Pour $q \neq 1$, la somme des termes consécutifs vaut :
\[
\boxed{S = (\text{premier terme}) \times \frac{1 - q^{\,\text{nombre de termes}}}{1 - q}}
\]
En particulier : $\displaystyle \sum_{k=0}^{n} u_k = u_0 \,\frac{1 - q^{\,n+1}}{1 - q}$.
\end{theoreme}

\begin{exemple}
$1 + 2 + 4 + 8 + \dots + 2^{10} = 1 \times \dfrac{1 - 2^{11}}{1 - 2} = 2^{11} - 1 = 2047$.
\end{exemple}

\begin{appliindus}[-- Intérêts composés et dépréciation]
Un capital placé à $3\,\%$ par an suit une suite géométrique de raison $q = 1{,}03$ : $C_n = C_0 \times 1{,}03^{\,n}$. À l'inverse, la valeur d'une machine qui perd $15\,\%$ par an suit une suite géométrique de raison $q = 0{,}85$. Ces modèles servent au calcul d'amortissement comptable et de retour sur investissement.
\end{appliindus}

\subsection{Comportement à l'infini}

\begin{theoreme}
Pour une suite géométrique de raison $q > 0$ :
\begin{itemize}
  \item si $q > 1$ : la suite croît vers $+\infty$ (croissance exponentielle) ;
  \item si $q = 1$ : la suite est constante ;
  \item si $0 < q < 1$ : la suite décroît vers $0$.
\end{itemize}
\end{theoreme}

\section{Recherche d'un seuil \hfill {\normalsize\color{uimmred}[C03]}}

Déterminer un \textbf{seuil}, c'est trouver le plus petit rang $n$ à partir duquel une condition est vérifiée (par exemple $u_n \geq a$ ou $u_n \leq 10^{-p}$). On utilise un tableur ou un algorithme avec une boucle conditionnelle.

\begin{methode}[-- Principe de l'algorithme de seuil]
\begin{enumerate}
  \item Initialiser le premier terme et le rang $n = 0$ ;
  \item Tant que la condition n'est pas atteinte : calculer le terme suivant et incrémenter $n$ ;
  \item Afficher le rang $n$ qui réalise le seuil.
\end{enumerate}
\end{methode}

\begin{lstlisting}[style=uimmpython, caption={Plus petit rang n tel que u(n) depasse un seuil}]
# Suite geometrique : u0 = 1000 ; q = 1.05 ; seuil a = 2000
u = 1000
q = 1.05
seuil = 2000
n = 0
while u < seuil:
    u = q * u
    n = n + 1
print("Premier rang depassant le seuil : n =", n)
print("Valeur correspondante : u =", round(u, 2))
\end{lstlisting}

\begin{exemple}[-- Lecture au tableur]
Avec $u_0 = 1000$ et $q = 1{,}05$, on recopie la formule \texttt{=1,05*B2} vers le bas et on repère la première ligne où la valeur dépasse $2000$. On trouve $n = 15$ ($u_{15} \approx 2079$).
\end{exemple}

\begin{appliindus}[-- Maintenance préventive]
Le jeu d'un roulement augmente de $4\,\%$ à chaque cycle de production. En partant d'un jeu initial mesuré, l'algorithme de seuil donne le nombre de cycles avant d'atteindre le jeu maximal toléré : c'est la base d'un plan de maintenance préventive.
\end{appliindus}

\section{Exercices d'application}

\subsection*{Exercice 1 -- Calcul de termes \hfill {\normalsize\color{uimmred}[C01]}}

  \item $u_n = 4n - 3$ : calculer $u_0$, $u_5$ et $u_{20}$.
  \item $u_0 = 1$, $u_{n+1} = 3u_n - 2$ : calculer $u_1$, $u_2$, $u_3$.
\end{enumerate}

\subsection*{Exercice 2 -- Nature d'une suite \hfill {\normalsize\color{uimmred}[C01]}}
Pour chaque suite, dire si elle est arithmétique, géométrique ou ni l'une ni l'autre, en précisant la raison.
\begin{enumerate}[label=\alph*)]
  \item $3,\; 7,\; 11,\; 15,\dots$
  \item $2,\; 6,\; 18,\; 54,\dots$
  \item $1,\; 4,\; 9,\; 16,\dots$
\end{enumerate}

\subsection*{Exercice 3 -- Suite arithmétique \hfill {\normalsize\color{uimmred}[C01]\,[C02]}}
Une suite arithmétique a pour premier terme $u_0 = 7$ et raison $r = 4$.
\begin{enumerate}[label=\alph*)]
  \item Donner l'expression de $u_n$.
  \item Calculer $u_{25}$.
  \item Calculer la somme $u_0 + u_1 + \dots + u_{25}$.
\end{enumerate}

\subsection*{Exercice 4 -- Suite géométrique \hfill {\normalsize\color{uimmred}[C01]\,[C02]}}
Une population de bactéries double toutes les heures. À $t = 0$, on compte $500$ bactéries.
\begin{enumerate}[label=\alph*)]
  \item Modéliser par une suite géométrique : donner $u_0$, $q$ et $u_n$.
  \item Combien de bactéries après $6$\,heures ?
  \item Calculer le nombre total cumulé sur les $6$ premières heures.
\end{enumerate}

\subsection*{Exercice 5 -- Recherche de seuil \hfill {\normalsize\color{uimmred}[C03]}}
Un capital de $5000\,$€ est placé à $2{,}5\,\%$ par an.
\begin{enumerate}[label=\alph*)]
  \item Exprimer le capital $C_n$ après $n$ années.
  \item Écrire un algorithme déterminant l'année où le capital dépasse $7000\,$€.
